# Postprod AI worker dependencies.
# Do NOT install packages that talk to external APIs other than the
# in-cluster app and vLLM Service.
#
# CUDA pinning: the GPU nodes run driver 550.x (A100), which supports
# CUDA 12.x up to 12.4 but NOT CUDA 13. The NGC base (24.10) ships an
# *alpha* torch 2.5 and NO torchaudio, so pip pulled torchaudio from
# PyPI — and the latest (2.11.0) is a CUDA-13 build, crashing the worker
# with `OSError: libcudart.so.13: cannot open shared object file`.
# Fix: install a single, self-consistent torch/torchvision/torchaudio
# trio from the CUDA-12.4 wheel index, overriding the NGC alpha torch.
# These wheels bundle the CUDA-12 runtime (libcudart.so.12) which the
# 550 driver runs natively. ctranslate2 (faster-whisper backend) stays
# on its CUDA-12 build. Bump cu124→cu126/cu128 only after the GPU node
# driver is bumped to 565+/580+.
--extra-index-url https://download.pytorch.org/whl/cu124
torch==2.12.0
torchvision==0.21.0
torchaudio==2.6.0
# pyannote / transformers behave best on numpy 1.x; avoid the 2.x ABI flip.
numpy<2

# HTTP client — used to talk to the app's /api/internal/* endpoints
# and to the in-cluster vLLM service. Sync API is plenty here: a
# worker is sequential by design (one job per Pod).
httpx==0.27.2

# Pydantic — validates the claim response shape so we don't deref
# `None`s on a malformed payload. Matches what `lib/ai/schemas.ts`
# enforces server-side.
pydantic==2.9.2

# ASR + diarization. WhisperX bundles faster-whisper + pyannote
# integration + word-level alignment via wav2vec2.
# Pin a 3.3.x (compatible with pyannote 3.3) — whisperx 3.1.x bloccava
# pyannote a 3.1.1 (vecchia, alcune voci IT mal allineate).
# Pyannote stesso lo lasciamo libero: whisperx tira la versione
# compatibile per sua dipendenza.
whisperx>=3.3.0,<3.5.0

# CTranslate2 backend for faster-whisper. Left to whisperx to resolve
# transitively (it hard-pins faster-whisper to a CT2 4.4.x build, which
# links cuDNN 8). Bumping CT2 to a cuDNN-9 build (>=4.5) is impossible
# without dropping the whisperx<3.5 cap (needed for pyannote 3.3 align).
# So instead the cuDNN-8 runtime libs (libcudnn_*.so.8) are provided in
# the image on a side LD_LIBRARY_PATH dir — see Dockerfile.worker — since
# the base ships cuDNN 9 only. cuDNN 8 and 9 coexist (distinct sonames).

# Subtitle / VTT writing. Custom helpers in worker/vtt.py use this.
webvtt-py==0.5.1

# Storage upload uses presigned PUT URLs — no SDK needed.

# Tokenizer for prompt construction (used by LLM client).
# Bump >=4.48.0 per CVE-2024-11392 (deserialization untrusted data,
# HIGH severity rilevato da Trivy nel CI dell'app — anche se l'uso qui
# è solo per costruire prompt, e non scarichiamo modelli arbitrari,
# evita il blocco del security scan).
transformers==4.48.3

# TTS — Piper (rhasspy/piper-tts). MIT license, voci pre-trained
# scaricate da huggingface.co/rhasspy/piper-voices. Gira su CPU
# (ONNX runtime), niente GPU richiesta. Per DUB job.
piper-tts==1.2.0

# Sentry / structured logging optional — we use stdlib logging
# configured for JSON. Cluster Loki picks them up.
gender-guesser>=0.4,<1

# AudioSeal (Meta, MIT) — watermark inudibile sull'audio doppiato per
# conformità AI Act Art. 50. Gira sul torch già pinnato (2.6 cu124); il
# trio torch/torchvision/torchaudio resta fissato sopra (pin esatti
# vincono sulle dipendenze >= di audioseal). Modello scaricato da HF a
# runtime (audioseal_wm_16bits, ~pochi MB). Best-effort: il DUB non
# fallisce se il watermark non riesce (vedi worker/tts.py).
audioseal==0.1.4
